    {
         ( "Sales Amount ", [Sales Amount] ),

            [Sales Amount],
            ALL ( 'Product'[Color] )
        ) ),

            [Sales Amount],
            ALL ( 'Product' )
        ) ),

            [Sales Amount],
            ALL ()
        ) ),

            [Sales Amount],
            ALL ( Sales )
        ) )
    },
    'Product'[Color] = "Red",
